# 함수 포인터 계산

1. 함수 포인터를 사용한 간단한 계산

```
#include <iostream>

using namespace std;

using calculate = int(*)(int a, int b);

int Add(int a, int b) { return a + b; }
int Subtraction(int a, int b) { return a - b; }

int main() 
{

	calculate cal[2] = { &Add, &Subtraction };

	int n, m, num;

	cout << "원하는 값을 넣어주세요: ";
	cin >> n >> m;
	
	cout << "원하는 계산을 선택해 주세요"<< endl;
	cout << "1. 더하기" << endl;
	cout << "2. 빼기" << endl;
	cin >> num;

	switch (num)
	{
	case 1:
	{
		cout << Add(n, m);
		break;
	}
	case 2:
	{
		cout << Subtraction(n, m);
		break;
	}
	default:
		break;
	}

	return 0;	
}
```
